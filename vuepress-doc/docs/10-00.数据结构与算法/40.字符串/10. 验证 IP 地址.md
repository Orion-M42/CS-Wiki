## 🔔 解题思路

没什么好说的，边界情况（corner case）比较多

另外，或 `|`，点 `.`，加 `+`，乘 `*`，在字符串中出现时，如果这个字符串需要被 split，则 split 时候，需要在前面加两个反斜杠 `\\` 进行转义


```java
class Solution {
    private static final String IPV4 = "IPv4";
    private static final String IPv6 = "IPv6";
    private static final String NEITHER = "Neither";

    public String validIPAddress(String queryIP) {
        if (isIpv4(queryIP)) {
            return IPV4;
        }
        else if (isIpv6(queryIP)) {
            return IPv6;
        }

        return NEITHER;
    }

    private boolean isIpv4(String s) {
        if (s == null || s.length() == 0) {
            return false;
        }

        // 如果最后 n 位都是切割符，split(" ")不会继续切分，split(" ", -1) 会继续切分
        // 防止诸如 192.168.1.1.. 这种情况
        String[] str = s.split("\\.", -1);
        if (str.length != 4) {
            return false;
        }

        for (String item : str) {
            if (item.length() > 3 || item.length() < 1) {
                return false;
            }

            int sum = 0;

            // 前导 0
            if (item.length() > 1 && item.charAt(0) == '0') {
                return false;
            }

            for (char c : item.toCharArray()) {
                if (c > '9' || c < '0') {
                    return false;
                }

                sum = sum * 10 + c - '0';
                if (sum > 255 || sum < 0) {
                    return false;
                }
            }
        }

        return true;
    }

    private boolean isIpv6(String s) {
        if (s == null || s.length() == 0) {
            return false;
        }

        // 如果最后 n 位都是切割符，split(" ")不会继续切分，split(" ", -1) 会继续切分
        String[] str = s.split(":", -1);
        if (str.length != 8) {
            return false;
        }

        for (String item : str) {
            if (item.length() > 4 || item.length() < 1) {
                return false;
            }

            for (char c : item.toCharArray()) {
                if (!((c <= '9' && c >= '0') || (c <= 'F' && c >= 'A') || (c <= 'f' && c >= 'a'))) {
                    return false;
                }
            }
        }

        return true;
    }
}
```
